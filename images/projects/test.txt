    <script>
        // Get the font size control, value display, and all caption and column elements
        const fontSizeControl = document.getElementById('fontSizeControl');
        const fontSizeValue = document.getElementById('fontSizeValue');
        const elementsToResize = document.querySelectorAll('.carousel-caption h5, .carousel-caption p, .mt-8 h3, .mt-8 p');

        // Add an event listener to the slider
        fontSizeControl.addEventListener('input', (event) => {
            const newFontSize = event.target.value;
            // Update the display value
            fontSizeValue.textContent = `${newFontSize}px`;

            // Loop through all selected elements and update their font size
            elementsToResize.forEach(element => {
                element.style.fontSize = `${newFontSize}px`;
            });
        });
    </script>
